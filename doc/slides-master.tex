\begin{frame}
  \frametitle{Introducción}
  \framesubtitle{Motivación}

    \begin{itemize}
  \item

  \item
    Menos tiempo de experimentacion y analisis debido a la necesidad de optimización.
  \item
    El codigo optimizado puede ejecutarse ordenes de magnitud mejor que una implemnentación directa.
  \end{itemize}
  
\end{frame}
